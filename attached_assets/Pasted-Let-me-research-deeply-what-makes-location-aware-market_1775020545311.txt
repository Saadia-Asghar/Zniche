> ### Step 2 — Live currency conversion (Frankfurter API, 100% free, no key)
>
> ```javascript
> // Converts any USD amount to user's local currency
> const convertPrice = async (usdAmount, targetCurrency) => {
>   if (targetCurrency === 'USD') return { amount: usdAmount, symbol: '$', formatted: `$${usdAmount}` };
>   try {
>     const res = await fetch(
>       `https://api.frankfurter.dev/v2/latest?base=USD&symbols=${targetCurrency}`
>     );
>     const data = await res.json();
>     const rate = data.rates[targetCurrency];
>     const converted = (usdAmount * rate).toFixed(0);
>     const symbols = { PKR:'₨', INR:'₹', GBP:'£', EUR:'€', AUD:'A$', CAD:'C$', SAR:'﷼', AED:'د.إ', NGN:'₦', BDT:'৳', EGP:'£E', KES:'KSh', MYR:'RM', IDR:'Rp' };
>     const sym = symbols[targetCurrency] || targetCurrency + ' ';
>     return { amount: converted, symbol: sym, formatted: `${sym}${converted}`, rate };
>   } catch { return { amount: usdAmount, symbol: '$', formatted: `$${usdAmount}`, rate: 1 }; }
> };
> ```
>
> Cache exchange rates for 1 hour in `localStorage.zniche_rates`.

> ### Step 3 — PPP price recommendations (hardcoded ratios, no API needed)
>
> ```javascript
> // PPP adjustment factors relative to USA (1.0)
> // Lower number = less purchasing power = suggest lower price
> const PPP_FACTORS = {
>   US:1.00, GB:0.92, DE:0.88, FR:0.87, AU:0.86, CA:0.85,
>   NZ:0.82, JP:0.78, SG:0.75, AE:0.72, SA:0.68, KR:0.66,
>   MX:0.42, BR:0.40, ZA:0.38, TH:0.35, CN:0.58, RU:0.45,
>   TR:0.30, AR:0.25, EG:0.22, PH:0.28, VN:0.26, MY:0.38,
>   NG:0.18, PK:0.15, BD:0.16, IN:0.20, ID:0.24, KE:0.20,
>   GH:0.19, ET:0.12, TZ:0.14, UG:0.13
> };
>
> const getPPPPrice = (usdPrice, countryCode) => {
>   const factor = PPP_FACTORS[countryCode] || 0.60; // default for unlisted
>   const pppPrice = Math.round(usdPrice * factor);
>   // Round to psychologically clean number
>   if (pppPrice < 10) return Math.round(pppPrice);
>   if (pppPrice < 50) return Math.round(pppPrice / 5) * 5 - 1; // e.g. 14→14, 29→29
>   return Math.round(pppPrice / 10) * 10 - 1; // e.g. 49, 99, 197
> };
> ```
